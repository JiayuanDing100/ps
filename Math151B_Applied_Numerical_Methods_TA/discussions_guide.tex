	\begin{enumerate}
	\item Sample problem $y' = ye^t, y(0) = 1$
	\item Taylor expansion with 2 variables
	\item Big O notations
	\end{enumerate}

	\item Butcher tableau
	\item Stability = continuity + Lipschitz condition
	\end{enumerate}
